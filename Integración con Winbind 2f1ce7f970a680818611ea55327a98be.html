#!/bin/bash
# prapersam ./29_1_getent_passwd.sh | tee getentpasswd_prapersam.txt

printf "%-25s %-6s %-6s %-35s %s\n" "USUARIO" "UID" "GID" "HOME" "SHELL"
printf "%-25s %-6s %-6s %-35s %s\n" "-------" "---" "---" "----" "-----"

samba-tool user list | while read -r u; do

    # CERRAMOS STDIN PARA EVITAR BUCLE
    info=$(samba-tool user show "$u" </dev/null 2>/dev/null)

    uid=$(echo "$info"   | awk -F': ' '/uidNumber/ {print $2}')
    [ -z "$uid" ] && continue

    gid=$(echo "$info"   | awk -F': ' '/gidNumber/ {print $2}')
    home=$(echo "$info"  | awk -F': ' '/unixHomeDirectory/ {print $2}')
    shell=$(echo "$info" | awk -F': ' '/loginShell/ {print $2}')

    [ -z "$gid" ]   && gid="$uid"
    [ -z "$home" ]  && home="/home/$u"
    [ -z "$shell" ] && shell="/bin/false"

    printf "%-25s %-6s %-6s %-35s %s\n" "$u" "$uid" "$gid" "$home" "$shell"

done

#!/bin/bash
# prapersam

printf "%-25s %-8s %s\n" "GRUPO" "GID" "MIEMBROS"
printf "%-25s %-8s %s\n" "-----" "---" "--------"

for g in $(samba-tool group list | grep '^g-'); do
    # Obtener GID (RFC2307)
    gid=$(samba-tool group show "$g" 2>/dev/null | awk '/gidNumber/ {print $2}')

    # Si no tiene gidNumber, lo marcamos
    [ -z "$gid" ] && gid="N/A"

    # Obtener miembros reales del AD
    members=$(samba-tool group listmembers "$g" 2>/dev/null | tr '\n' ',' | sed 's/,$//')

    [ -z "$members" ] && members="(ninguno)"

    printf "%-25s %-8s %s\n" "$g" "$gid" "$members"
done
